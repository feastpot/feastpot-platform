SECTION 3 — Delivery slot visual calendar:

Create apps/web/src/components/checkout/slot-picker.tsx:

'use client'
import { useState } from 'react'
import { cn } from '@/lib/utils'
import { format, addDays, isToday, isTomorrow } from 'date-fns'

interface SlotPickerProps {
  availableDays: number[]  // 0=Sun, 1=Mon … 6=Sat
  slotOpenTime:  string    // "11:00"
  slotCloseTime: string    // "18:00"
  leadTimeHours: number
  maxAdvanceDays: number
  value: Date | null
  onChange: (date: Date) => void
}

export function SlotPicker({ availableDays, slotOpenTime, slotCloseTime, leadTimeHours, maxAdvanceDays, value, onChange }: SlotPickerProps) {
  const [selectedDate, setSelectedDate] = useState<Date | null>(null)

  // Build available dates
  const today = new Date()
  const cutoff = new Date(today.getTime() + leadTimeHours * 60 * 60 * 1000)
  const dates: Date[] = []
  for (let i = 0; i <= maxAdvanceDays; i++) {
    const d = addDays(today, i)
    if (availableDays.includes(d.getDay()) && d > cutoff) dates.push(d)
  }

  // Build time slots for selected date (1-hour slots between open and close)
  const generateSlots = (date: Date) => {
    const [openH, openM]   = slotOpenTime.split(':').map(Number)
    const [closeH, closeM] = slotCloseTime.split(':').map(Number)
    const slots: string[] = []
    for (let h = openH; h < closeH; h++) {
      slots.push(`${h.toString().padStart(2,'0')}:00–${(h+1).toString().padStart(2,'0')}:00`)
    }
    return slots
  }

  const formatDateLabel = (d: Date) => {
    if (isToday(d)) return 'Today'
    if (isTomorrow(d)) return 'Tomorrow'
    return format(d, 'EEE d MMM')
  }

  return (
    <div>
      {/* Date row */}
      <div className="flex gap-2 overflow-x-auto scrollbar-hide pb-2">
        {dates.length === 0 ? (
          <p className="text-sm text-mid py-2">No available slots — this vendor needs {leadTimeHours}h notice.</p>
        ) : dates.map(d => (
          <button
            key={d.toISOString()}
            onClick={() => setSelectedDate(d)}
            className={cn(
              'flex-shrink-0 flex flex-col items-center px-3 py-2.5 rounded-xl border text-sm font-medium transition-all',
              selectedDate?.toDateString() === d.toDateString()
                ? 'bg-teal text-white border-teal'
                : 'bg-white text-dark border-gray-200 hover:border-teal/50'
            )}
          >
            <span className="text-[10px] opacity-70">{format(d, 'EEE')}</span>
            <span className="text-base font-bold">{format(d, 'd')}</span>
            <span className="text-[10px] opacity-70">{format(d, 'MMM')}</span>
          </button>
        ))}
      </div>

      {/* Time slot pills */}
      {selectedDate && (
        <div className="flex flex-wrap gap-2 mt-3">
          {generateSlots(selectedDate).map(slot => (
            <button
              key={slot}
              onClick={() => {
                const [h] = slot.split(':')
                const d = new Date(selectedDate)
                d.setHours(parseInt(h), 0, 0, 0)
                onChange(d)
              }}
              className={cn(
                'px-3 py-2 rounded-xl text-sm font-medium border transition-all',
                value && format(value, 'HH:mm') === slot.split('–')[0]
                  ? 'bg-brand text-white border-brand'
                  : 'bg-white text-dark border-gray-200 hover:border-brand/50'
              )}
            >
              {slot}
            </button>
          ))}
        </div>
      )}
    </div>
  )
}
