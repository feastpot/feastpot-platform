import { IsBoolean, IsOptional, IsString, IsDateString } from 'class-validator'

export class SetAcceptingOrdersDto {
  @IsBoolean()
  acceptingOrders: boolean

  @IsOptional() @IsDateString()
  pausedUntil?: string  // ISO timestamp — auto-resume

  @IsOptional() @IsString()
  pauseReason?: string
}

File: apps/api/src/modules/vendors/vendors.service.ts

async setAcceptingOrders(userId: string, dto: SetAcceptingOrdersDto) {
  const vendor = await this.prisma.vendor.findFirst({ where: { userId } })
  if (!vendor) throw new NotFoundException()

  return this.prisma.vendor.update({
    where: { id: vendor.id },
    data: {
      acceptingOrders: dto.acceptingOrders,
      pausedUntil:     dto.pausedUntil ? new Date(dto.pausedUntil) : null,
      pauseReason:     dto.pauseReason ?? null,
    },
  })
}

STEP 3 — Cron to auto-resume paused vendors
File: apps/api/src/modules/vendors/vendors-cron.service.ts

@Cron('*/15 * * * *')  // every 15 minutes
async autoResumeVendors() {
  const expired = await this.prisma.vendor.findMany({
    where: {
      acceptingOrders: false,
      pausedUntil:     { lte: new Date() },
    },
  })
  for (const vendor of expired) {
    await this.prisma.vendor.update({
      where: { id: vendor.id },
      data: { acceptingOrders: true, pausedUntil: null, pauseReason: null },
    })
    this.logger.log(`[VendorCron] Auto-resumed ${vendor.businessName}`)
  }
}
