import { Resend } from 'resend'
import { Injectable } from '@nestjs/common'
import { ConfigService } from '@nestjs/config'

@Injectable()
export class ResendProvider {
  private readonly resend: Resend
  constructor(private config: ConfigService) {
    this.resend = new Resend(this.config.get('RESEND_API_KEY'))
  }
  async send(to: string, subject: string, html: string): Promise<void> {
    const { error } = await this.resend.emails.send({
      from: 'Feastpot <noreply@feastpot.co.uk>',
      to, subject, html
    })
    if (error) throw new Error(`Resend error: ${error.message}`)
  }
}

For Twilio provider — confirm or create apps/api/src/modules/notifications/providers/twilio.provider.ts:
import Twilio from 'twilio'
import { Injectable } from '@nestjs/common'
import { ConfigService } from '@nestjs/config'

@Injectable()
export class TwilioProvider {
  private readonly client: Twilio.Twilio
  private readonly from: string
  constructor(private config: ConfigService) {
    this.client = Twilio(
      this.config.get('TWILIO_ACCOUNT_SID'),
      this.config.get('TWILIO_AUTH_TOKEN')
    )
    this.from = this.config.get('TWILIO_FROM_NUMBER') ?? ''
  }
  async sendSms(to: string, body: string): Promise<void> {
    if (!this.from) { console.warn('TWILIO_FROM_NUMBER not set — SMS skipped'); return }
    await this.client.messages.create({ from: this.from, to, body })
  }
}

  const { to, template, data } = job.data
  let subject = '', html = ''

    case 'order_confirmation':    ({ subject, html } = orderConfirmationTemplate(data)); break
    case 'order_accepted':        ({ subject, html } = orderAcceptedTemplate(data)); break
    case 'order_dispatched':      ({ subject, html } = orderDispatchedTemplate(data)); break
    case 'order_delivered':       ({ subject, html } = orderDeliveredTemplate(data)); break
    case 'review_request':        ({ subject, html } = reviewRequestTemplate(data)); break
    case 'payout_statement':      ({ subject, html } = payoutStatementTemplate(data)); break
    case 'document_expiry_warning': ({ subject, html } = documentExpiryTemplate(data)); break
    case 'dispute_opened':        ({ subject, html } = disputeOpenedTemplate(data)); break
    default:
      this.logger.warn(`Unknown email template: ${template}`)
      return
  }
  await this.resendProvider.send(to, subject, html)
  break

In the SEND_SMS job handler:
case 'send_sms': {
  const { phone, template, data } = job.data
  const fn = smsTemplates[template]
  if (!fn) { this.logger.warn(`Unknown SMS template: ${template}`); return }
  await this.twilioProvider.sendSms(phone, fn(data))
  break
}
