'use client'
import { useEffect, useState } from 'react'

export function SWUpdatePrompt() {
  const [show, setShow] = useState(false)
  useEffect(() => {
    if (!('serviceWorker' in navigator)) return
    navigator.serviceWorker.ready.then(reg => {
      reg.addEventListener('updatefound', () => {
        const w = reg.installing
        w?.addEventListener('statechange', () => {
          if (w.state === 'installed' && navigator.serviceWorker.controller) setShow(true)
        })
      })
    })
  }, [])
  if (!show) return null
  return (
    <div style={{ position: 'fixed', bottom: 'calc(72px + env(safe-area-inset-bottom))',
      left: '12px', right: '12px', zIndex: 100, background: '#1C1C1A',
      borderRadius: '14px', padding: '12px 14px',
      display: 'flex', alignItems: 'center', justifyContent: 'space-between' }}>
      <span style={{ color: 'white', fontSize: '13px' }}>Update available</span>
      <button onClick={() => window.location.reload()}
        style={{ background: '#E8520A', color: 'white', border: 'none',
          padding: '6px 14px', borderRadius: '8px', fontWeight: 600, fontSize: '12px', cursor: 'pointer' }}>
        Update now
      </button>
    </div>
  )
}
