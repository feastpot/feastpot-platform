export interface CoverageResult {
  covered:       boolean
  distanceMiles: number | null
  radiusMiles:   number
  message:       string
}

export async function checkDeliveryCoverage(
  customerPostcode: string,
  vendor: {
    deliveryConfig?: {
      latitude?:            number | null
      longitude?:           number | null
      deliveryRadiusMiles?: number | null
    } | null
  }
): Promise<CoverageResult> {
  const config = vendor.deliveryConfig

  // No delivery config: assume vendor delivers everywhere (early-stage default)
  if (!config || !config.latitude || !config.longitude) {
    return {
      covered:       true,
      distanceMiles: null,
      radiusMiles:   0,
      message:       'Delivery area not yet configured',
    }
  }

  const radiusMiles = Number(config.deliveryRadiusMiles ?? 5)

  // Geocode customer postcode
  let customerCoords: { lat: number; lng: number } | null = null
  try {
    const clean  = customerPostcode.replace(/\s+/g, '').toUpperCase()
    const cached = sessionStorage.getItem(`geocode:${clean}`)
    if (cached) {
      customerCoords = JSON.parse(cached)
    } else {
      const res  = await fetch(`https://api.postcodes.io/postcodes/${clean}`)
      if (res.ok) {
        const data = await res.json()
        if (data.result) {
          customerCoords = { lat: data.result.latitude, lng: data.result.longitude }
          sessionStorage.setItem(`geocode:${clean}`, JSON.stringify(customerCoords))
        }
      }
    }
  } catch {
    // Geocode failure: fail open (show vendor) rather than fail closed (hide vendor)
    return {
      covered:       true,
      distanceMiles: null,
      radiusMiles,
      message:       'Could not verify coverage — delivery availability shown at checkout',
    }
  }

  if (!customerCoords) {
    return {
      covered:       false,
      distanceMiles: null,
      radiusMiles,
      message:       'Enter a valid UK postcode to check delivery availability',
    }
  }

  // Haversine distance
  const R      = 3958.8
  const toRad  = (d: number) => (d * Math.PI) / 180
  const dLat   = toRad(Number(config.latitude) - customerCoords.lat)
  const dLng   = toRad(Number(config.longitude) - customerCoords.lng)
  const a      = Math.sin(dLat/2)**2 +
    Math.cos(toRad(customerCoords.lat)) *
    Math.cos(toRad(Number(config.latitude))) *
    Math.sin(dLng/2)**2
  const distanceMiles = R * 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1-a))

  const covered = distanceMiles <= radiusMiles

  return {
    covered,
    distanceMiles: Math.round(distanceMiles * 10) / 10,
    radiusMiles,
    message: covered
      ? `${(Math.round(distanceMiles * 10) / 10).toFixed(1)} miles away · Delivers to you`
      : `${(Math.round(distanceMiles * 10) / 10).toFixed(1)} miles away · Outside delivery area`,
  }
}

'use client'
import { useEffect, useState } from 'react'
import { checkDeliveryCoverage, type CoverageResult } from '@/lib/api/coverage'

export function CoverageBadge({
  postcode, vendor,
}: {
  postcode: string
  vendor: any
}) {
  const [result, setResult] = useState<CoverageResult | null>(null)

  useEffect(() => {
    if (!postcode) return
    checkDeliveryCoverage(postcode, vendor).then(setResult)
  }, [postcode, vendor.id])

  if (!result) return null

  return (
    <span style={{
      display:   'inline-flex',
      alignItems: 'center',
      gap:        '4px',
      fontSize:   '11px',
      fontWeight: 600,
      padding:    '3px 8px',
      borderRadius: '6px',
      background: result.covered ? '#E8F5EC' : '#FFF8EC',
      color:      result.covered ? '#1E7B34' : '#B87B00',
    }}>
      {result.covered ? '✓' : '⚠'} {result.message}
    </span>
  )
}
