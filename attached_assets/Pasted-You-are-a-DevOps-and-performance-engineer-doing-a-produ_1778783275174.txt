The generateMetadata function should:
export async function generateMetadata({ params }: { params: { slug: string } }): Promise<Metadata> {
  const vendor = await getVendor(params.slug)
  if (!vendor) return { title: 'Vendor not found | Feastpot' }

  const topDishes = vendor.menus?.flatMap(m=>m.items||[]).slice(0,3).map(i=>i.name).join(', ')

  return {
    title: `${vendor.businessName} | Feastpot`,
    description: vendor.description ||
      `${vendor.cuisineTypes?.join(' & ')} home cooking in ${vendor.address?.city || 'London'}. ${topDishes ? 'Dishes include: ' + topDishes + '.' : ''} Order party trays, frozen packs and more.`,
    openGraph: {
      title: vendor.businessName,
      description: vendor.description,
      images: vendor.coverUrl ? [{ url: vendor.coverUrl, width: 1200, height: 630 }] : [],
    },
  }
}
