STEP 1 — Resend: verify the sending domain
════════════════════════════════════════════════════════

OPERATIONAL (do in DNS provider and Resend dashboard):
1. Log in to resend.com → Domains → Add domain → feastpot.co.uk
2. Add the following DNS records at your DNS provider:
   TYPE  NAME                          VALUE
   TXT   resend._domainkey.feastpot.co.uk  (value from Resend dashboard)
   MX    send.feastpot.co.uk              feedback-smtp.eu-west-1.amazonses.com (or as shown)

3. Wait for verification (5-30 minutes)
4. Status must show: Verified ✓

5. Add to Replit deployment secrets:
   RESEND_API_KEY    = re_live_...   (from resend.com/api-keys)
   RESEND_FROM_EMAIL = hello@feastpot.co.uk
   RESEND_FROM_NAME  = Feastpot

CRITICAL CHECK: The API key must start re_live_ not re_test_.
If it starts re_test_, emails deliver but show as "Sent from Resend test"
and may land in spam.

const NOTIFICATION_VARS = [
  'RESEND_API_KEY',
  'RESEND_FROM_EMAIL',
  'WHATSAPP_ACCESS_TOKEN',
  'WHATSAPP_PHONE_NUMBER_ID',
]

const missingNotifications = NOTIFICATION_VARS.filter(k => !process.env[k])
if (missingNotifications.length > 0) {
  logger.warn(
    `[STARTUP] Notification credentials missing: ${missingNotifications.join(', ')}. ` +
    'Notifications will be silently dropped until these are set.'
  )
  // Do NOT exit — degraded notifications are better than no platform
  // But log clearly so ops knows notifications are broken
}

@Post('admin/test/notifications')
@Roles('admin')
async testNotifications(@Body() dto: { email: string; phone?: string }) {
  if (process.env.NODE_ENV === 'production') {
    throw new ForbiddenException('Use staging for notification tests')
  }

  const results: Record<string, any> = {}

  // Test email
  try {
    await this.notificationsService.sendEmail({
      to:      dto.email,
      subject: 'Feastpot notification test ✅',
      html:    '<p>If you received this, Resend is working correctly.</p>',
    })
    results.email = 'sent'
  } catch (e) {
    results.email = `failed: ${e.message}`
  }

  // Test WhatsApp (if phone provided)
  if (dto.phone) {
    try {
      await this.notificationsService.sendWhatsApp({
        to:      dto.phone,
        message: 'Feastpot test — if you received this, WhatsApp is working. 🎉',
      })
      results.whatsapp = 'sent'
    } catch (e) {
      results.whatsapp = `failed: ${e.message}`
    }
  }

  return results
}
