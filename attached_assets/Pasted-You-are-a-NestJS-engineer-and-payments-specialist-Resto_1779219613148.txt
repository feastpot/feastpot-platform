describe('Refund flow', () => {
  it('issues a Stripe refund when customer cancels a pending order', async () => {
    // Arrange
    const mockOrder = {
      id: 'order-123',
      status: 'pending',
      customerId: 'customer-123',
      payment: { stripePaymentIntentId: 'pi_test_123' },
    }
    mockPrisma.order.findUnique.mockResolvedValue(mockOrder)
    mockStripe.paymentIntents.cancel.mockResolvedValue({ status: 'canceled' })

    // Act
    await ordersService.customerCancel('order-123', 'customer-123', 'Changed my mind')

    // Assert
    expect(mockStripe.paymentIntents.cancel).toHaveBeenCalledWith('pi_test_123')
    expect(mockPrisma.order.update).toHaveBeenCalledWith(
      expect.objectContaining({
        data: expect.objectContaining({ status: 'cancelled' }),
      })
    )
  })

  it('issues a Stripe refund (not cancel) when customer cancels an accepted order', async () => {
    const mockOrder = {
      id: 'order-456',
      status: 'accepted',
      customerId: 'customer-123',
      payment: { stripePaymentIntentId: 'pi_test_456' },
    }
    mockPrisma.order.findUnique.mockResolvedValue(mockOrder)
    mockStripe.refunds.create.mockResolvedValue({ id: 're_test_123', status: 'succeeded' })

    await ordersService.customerCancel('order-456', 'customer-123', 'Changed my mind')

    expect(mockStripe.refunds.create).toHaveBeenCalledWith(
      expect.objectContaining({ payment_intent: 'pi_test_456' })
    )
  })

  it('blocks refund when order is already preparing', async () => {
    const mockOrder = { id: 'order-789', status: 'preparing', customerId: 'customer-123' }
    mockPrisma.order.findUnique.mockResolvedValue(mockOrder)

    await expect(
      ordersService.customerCancel('order-789', 'customer-123', 'Changed my mind')
    ).rejects.toThrow('already being prepared')
  })
})
