STEP 1 — Schema
File: prisma/schema.prisma

model NotificationPreference {
  id       String  @id @default(dbgenerated("gen_random_uuid()")) @db.Uuid
  userId   String  @db.Uuid @map("user_id")
  channel  String  // email, sms, whatsapp, push
  key      String  // order_confirmation, order_dispatched, marketing, etc.
  enabled  Boolean @default(true)

  user     User    @relation(fields: [userId], references: [id], onDelete: Cascade)

  @@unique([userId, channel, key])
  @@index([userId])
  @@map("notification_preferences")
}

Run: npx prisma migrate dev --name add_notification_preferences

STEP 2 — API endpoints
File: apps/api/src/modules/notifications/notification-preferences.controller.ts

@Controller({ version: '1', path: 'notification-preferences' })
@UseGuards(JwtAuthGuard)
export class NotificationPreferencesController {
  @Get()
  async getPreferences(@CurrentUser() user: JwtUser) {
    return this.prefsService.getPreferences(user.id)
  }

  @Patch()
  async updatePreferences(
    @CurrentUser() user: JwtUser,
    @Body() dto: UpdatePreferencesDto,
  ) {
    return this.prefsService.updatePreferences(user.id, dto)
  }
}
