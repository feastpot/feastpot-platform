'use client'
import { useEffect, useState } from 'react'
import { cn } from '@/lib/utils'

interface StatCardProps {
  label:    string
  value:    string | number
  prefix?:  string  // e.g. "£"
  suffix?:  string  // e.g. " orders"
  change?:  number  // % vs last week, positive=good
  icon:     string  // emoji
  color:    'brand' | 'teal' | 'vendor' | 'gray'
}

export function StatCard({ label, value, prefix, suffix, change, icon, color }: StatCardProps) {
  const [displayed, setDisplayed] = useState(0)
  const numValue = typeof value === 'number' ? value : parseFloat(value.toString().replace(/[^0-9.]/g, ''))

  useEffect(() => {
    let start = 0
    const duration = 800
    const startTime = performance.now()
    const animate = (now: number) => {
      const progress = Math.min((now - startTime) / duration, 1)
      const eased = 1 - Math.pow(1 - progress, 3) // ease-out cubic
      setDisplayed(Math.round(eased * numValue))
      if (progress < 1) requestAnimationFrame(animate)
    }
    requestAnimationFrame(animate)
  }, [numValue])

  const colorMap = {
    brand:  'bg-brand-light border-brand/20 text-brand',
    teal:   'bg-teal-light border-teal/20 text-teal',
    vendor: 'bg-blue-50 border-blue-200 text-vendor',
    gray:   'bg-surface border-gray-200 text-mid',
  }

  return (
    <div className={cn('fp-card p-4 border', colorMap[color])}>
      <div className="text-2xl mb-1">{icon}</div>
      <div className="text-[22px] font-black text-dark">
        {prefix}{displayed}{suffix}
      </div>
      <div className="text-xs text-mid mt-0.5">{label}</div>
      {change !== undefined && (
        <div className={cn('text-[11px] font-medium mt-1', change >= 0 ? 'text-teal' : 'text-red-500')}>
          {change >= 0 ? '↑' : '↓'} {Math.abs(change)}% vs last week
        </div>
      )}
    </div>
  )
}

COUNTDOWN TIMER component (create apps/vendor/src/components/orders/countdown.tsx):

'use client'
import { useState, useEffect } from 'react'
import { cn } from '@/lib/utils'

export function Countdown({ expiresAt }: { expiresAt: Date }) {
  const [remaining, setRemaining] = useState(0)

  useEffect(() => {
    const tick = () => {
      const diff = Math.max(0, expiresAt.getTime() - Date.now())
      setRemaining(Math.floor(diff / 1000))
    }
    tick()
    const interval = setInterval(tick, 1000)
    return () => clearInterval(interval)
  }, [expiresAt])

  const mins = Math.floor(remaining / 60)
  const secs = remaining % 60
  const isUrgent = remaining < 180 // < 3 minutes
  const isAlmostExpired = remaining < 60 // < 1 minute

  return (
    <div className={cn(
      'flex items-center gap-2 px-3 py-1.5 rounded-xl text-sm font-bold transition-colors',
      isAlmostExpired ? 'bg-red-100 text-red-600 animate-pulse' :
      isUrgent        ? 'bg-amber-100 text-amber-700' :
                        'bg-surface text-mid'
    )}>
      ⏱️
      <span>
        {mins.toString().padStart(2,'0')}:{secs.toString().padStart(2,'0')}
      </span>
      {isUrgent && <span className="text-[10px] font-medium">Accept now</span>}
    </div>
  )
}
