You are a Next.js 15 engineer. Implement the vendor acquisition flow
entirely within the customer PWA at feastpot.co.uk.

ARCHITECTURE:
- All "Become a cook" / "Join FeastPot" / "Register interest" links
  point to feastpot.co.uk/become-a-vendor (customer PWA — apps/web)
- The interest form lives at feastpot.co.uk/become-a-vendor
- Form submission hits POST /v1/vendors/register-interest (existing API endpoint)
- NO redirect to vendor.feastpot.co.uk ever happens from any public page
- vendor.feastpot.co.uk is only reached via the approval invite email
- The vendor portal URL must not appear in any link, footer, sitemap, or
  robots.txt allow rule on the customer site

This page has two sections rendered on the same URL:
  Section A: Landing/marketing content (default view)

Use a local state toggle — no URL change, no routing between them.
The form is revealed inline on the same page with a smooth scroll.

    instagram: '', foodStory: '', hasFSA: '',

  const handleRegisterClick = () => {
    setShowForm(true)

      formRef.current?.scrollIntoView({ behavior: 'smooth', block: 'start' })
    }, 80)
  }

  const validate = () => {
    const e: Record<string, string> = {}
    if (!form.fullName.trim())                   e.fullName    = 'Required'
    if (!form.kitchenName.trim())                e.kitchenName = 'Required'
    if (!form.email.includes('@'))               e.email       = 'Enter a valid email'
    if (form.phone.replace(/\D/g,'').length < 10) e.phone      = 'Enter a valid phone number'
    if (!form.postcode.trim())                   e.postcode    = 'Required'
    if (!form.cuisineType ||
        form.cuisineType === 'Select cuisine type') e.cuisineType = 'Select a cuisine type'
    if (form.foodStory.trim().length < 20)       e.foodStory   = 'Tell us a little more (min 20 characters)'
    if (!form.hasFSA)                            e.hasFSA      = 'Please answer this question'
    if (!form.terms)                             e.terms       = 'You must accept the terms to continue'
    setErrors(e)
    return Object.keys(e).length === 0
  }

    if (!validate()) return
    setLoading(true)
    setServerError('')
    try {
      const res = await fetch(
        `${process.env.NEXT_PUBLIC_API_URL}/v1/vendors/register-interest`,
        {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify(form),
        }
      )
      if (!res.ok) {
        const data = await res.json().catch(() => ({}))
        setServerError(data.message || 'Something went wrong. Please try again.')
        return
      }
      setSubmitted(true)
      window.scrollTo({ top: 0, behavior: 'smooth' })
    } catch {
      setServerError('Could not submit — check your connection and try again.')
    } finally {
      setLoading(false)
    }

                        onClick={() => setForm(f => ({ ...f, kitchenType: kt }))}
                        style={{
                          padding: '11px 8px', border: 'none',
                          borderRight: i < 2 ? '1px solid #E8E8E8' : 'none',
                          background: form.kitchenType === kt ? '#1E7B34' : 'white',
                          color:      form.kitchenType === kt ? 'white' : '#1A1A1A',
                          fontSize: '13px', fontWeight: form.kitchenType === kt ? 700 : 400,
                          cursor: 'pointer', fontFamily: 'DM Sans, sans-serif',
                          transition: 'all .15s',
                        }}>
                        {kt}
                      </button>
                    ))}
                  </div>
                </div>

                {inp('Instagram handle', form.instagram,
                  v => setForm(f => ({ ...f, instagram: v })),
                  { placeholder: '@yourhandle', optional: true })}

                {/* Food story */}
                <div style={{ marginBottom: errors.foodStory ? '4px' : '14px' }}>
                  <label style={{
                    display: 'block', fontSize: '13px', fontWeight: 600,
                    color: '#1A1A1A', marginBottom: '6px',
                  }}>Tell us about your food</label>
                  <textarea
                    value={form.foodStory} rows={4}
                    onChange={e => setForm(f => ({ ...f, foodStory: e.target.value }))}
                    placeholder="Tell us about your dishes, specialties and cooking experience..."
                    style={{
                      width: '100%', padding: '12px 14px', borderRadius: '8px',
                      border: `1.5px solid ${errors.foodStory ? '#E8220A' : '#E8E8E8'}`,
                      fontSize: '14px', fontFamily: 'DM Sans, sans-serif',
                      outline: 'none', resize: 'vertical', color: '#1A1A1A',
                      boxSizing: 'border-box',
                    }}
                  />
                  {errors.foodStory && (
                    <p style={{
                      fontSize: '12px', color: '#E8220A',
                      marginTop: '4px', marginBottom: '6px',
                    }}>{errors.foodStory}</p>
                  )}
                </div>

                {/* FSA radio */}
                <div style={{ marginBottom: '14px' }}>
                  <label style={{
                    display: 'block', fontSize: '13px', fontWeight: 600,
                    color: '#1A1A1A', marginBottom: '8px',
                  }}>
                    Do you have food registration / hygiene details?
                  </label>
                  <div style={{ display: 'flex', gap: '24px' }}>
                    {['Yes', 'No'].map(opt => (
                      <label key={opt} style={{
                        display: 'flex', alignItems: 'center',
                        gap: '8px', cursor: 'pointer',
                      }}>
                        <input type="radio" name="hasFSA" value={opt}
                          checked={form.hasFSA === opt}
                          onChange={() => setForm(f => ({ ...f, hasFSA: opt }))}
                          style={{ accentColor: '#1E7B34', width: '16px', height: '16px' }} />
                        <span style={{ fontSize: '14px', color: '#1A1A1A' }}>{opt}</span>
                      </label>
                    ))}
                  </div>
                  {errors.hasFSA && (
                    <p style={{
                      fontSize: '12px', color: '#E8220A', marginTop: '4px',
                    }}>{errors.hasFSA}</p>
                  )}
                </div>

                {/* T&C */}
                <label style={{
                  display: 'flex', alignItems: 'flex-start', gap: '10px',
                  marginBottom: errors.terms ? '4px' : '20px', cursor: 'pointer',
                }}>
                  <input type="checkbox" checked={form.terms}
                    onChange={e => setForm(f => ({ ...f, terms: e.target.checked }))}
                    style={{
                      marginTop: '2px', accentColor: '#1E7B34',
                      width: '16px', height: '16px', flexShrink: 0,
                    }} />
                  <span style={{ fontSize: '13px', color: '#666666', lineHeight: 1.5 }}>
                    I agree to FeastPot's{' '}
                    <Link href="/legal/terms" target="_blank"
                      style={{ color: '#1E7B34', fontWeight: 600 }}>
                      Terms of Service
                    </Link>
                    {' '}and{' '}
                    <Link href="/legal/privacy" target="_blank"
                      style={{ color: '#1E7B34', fontWeight: 600 }}>
                      Privacy Policy
                    </Link>
                  </span>
                </label>
                {errors.terms && (
                  <p style={{
                    fontSize: '12px', color: '#E8220A', marginBottom: '14px',
                  }}>{errors.terms}</p>
                )}

                {/* Submit */}
                <button onClick={handleSubmit} disabled={loading}
                  style={{
                    width: '100%', padding: '15px', borderRadius: '8px', border: 'none',
                    background: loading ? '#E8E8E8' : '#1E7B34', color: 'white',
                    fontSize: '15px', fontWeight: 700,
                    cursor: loading ? 'default' : 'pointer',
                    fontFamily: 'DM Sans, sans-serif', transition: 'background .15s',
                  }}>
                  {loading ? 'Submitting…' : 'Submit interest'}
                </button>

                <p style={{
                  textAlign: 'center', fontSize: '13px', color: '#999999',
                  marginTop: '12px',
                  display: 'flex', alignItems: 'center',
                  justifyContent: 'center', gap: '6px',
                }}>
                  ⏱ We'll be in touch within 1–2 business days.
                </p>
              </div>
            </div>
          </div>

          {/* Hide sidebar on mobile */}
          <style>{`
            @media (max-width: 640px) {
              .vendor-form-sidebar { display: none !important; }
            }
          `}</style>
        </section>
      )}
    </div>
  )
}
