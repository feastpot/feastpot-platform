'use client'

import { useEffect, useState, useCallback } from 'react'
import { createBrowserClient } from '@supabase/ssr'

export function useOrderTracking(orderId: string, initialStatus: string) {
  const [status, setStatus] = useState(initialStatus)
  const [connected, setConnected] = useState(false)
  const supabase = createBrowserClient(
    process.env.NEXT_PUBLIC_SUPABASE_URL!,
    process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!
  )

  useEffect(() => {
    const channel = supabase
      .channel(`order-tracking-${orderId}`)
      .on(
        'postgres_changes',
        {
          event: 'UPDATE',
          schema: 'public',
          table: 'orders',
          filter: `id=eq.${orderId}`,
        },
        (payload) => {
          const newStatus = payload.new.status
          setStatus(newStatus)
          // Play a subtle sound on status change
          if (typeof window !== 'undefined') {
            new Audio('/sounds/status-update.mp3').play().catch(() => {})
          }
        }
      )
      .subscribe((status) => {
        setConnected(status === 'SUBSCRIBED')
      })

    return () => {
      supabase.removeChannel(channel)
    }
  }, [orderId])

  return { status, connected }
}

STEP 3 — Fallback polling

Add a polling fallback in case Realtime is unavailable (e.g. corporate firewalls block WebSockets):

In the useOrderTracking hook, add:

useEffect(() => {
  if (connected) return // skip polling if Realtime is connected
  
  const interval = setInterval(async () => {
    const response = await fetch(`/api/orders/${orderId}/status`)
    const data = await response.json()
    if (data.status !== status) setStatus(data.status)
  }, 30_000) // poll every 30 seconds
  
  return () => clearInterval(interval)
}, [connected, orderId, status])
