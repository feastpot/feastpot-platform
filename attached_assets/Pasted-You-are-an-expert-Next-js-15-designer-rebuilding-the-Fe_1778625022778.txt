'use client'
import { useState, useEffect } from 'react'
import { cn } from '@/lib/utils'

const CUISINES = ['Jollof Rice', 'Egusi Soup', 'Pepper Soup', 'Oxtail', 'Party Trays', 'Jerk Chicken', 'Pounded Yam', 'Small Chops']

export function AnimatedHeadline() {
  const [idx, setIdx] = useState(0)
  const [visible, setVisible] = useState(true)

  useEffect(() => {
    const timer = setInterval(() => {
      setVisible(false)
      setTimeout(() => {
        setIdx(i => (i + 1) % CUISINES.length)
        setVisible(true)
      }, 300)
    }, 2200)
    return () => clearInterval(timer)
  }, [])

  return (
    <div className="text-center">
      <h1 className="text-[28px] font-black leading-tight text-white">
        Authentic
      </h1>
      <div className="h-10 flex items-center justify-center overflow-hidden">
        <span
          className={cn(
            'text-[28px] font-black text-white transition-all duration-300',
            visible ? 'opacity-100 translate-y-0' : 'opacity-0 translate-y-2'
          )}
        >
          {CUISINES[idx]}
        </span>
      </div>
      <h1 className="text-[28px] font-black leading-tight text-white">
        delivered in London
      </h1>
    </div>
  )
}

Create apps/web/src/components/home/postcode-hero.tsx:

'use client'
import { useState } from 'react'
import { useRouter } from 'next/navigation'
import { MapPin, ArrowRight } from 'lucide-react'
import { AnimatedHeadline } from './animated-headline'

export function PostcodeHero() {
  const [postcode, setPostcode] = useState('')
  const [error, setError] = useState('')
  const router = useRouter()

  const validate = (v: string) => /^[A-Z]{1,2}[0-9][0-9A-Z]?s?[0-9][A-Z]{2}$/i.test(v.trim())

  const handleSubmit = () => {
    if (!validate(postcode)) {
      setError('Please enter a valid UK postcode')
      return
    }
    localStorage.setItem('fp.postcode', postcode.toUpperCase().trim())
    router.push(`/vendors?postcode=${encodeURIComponent(postcode.toUpperCase().trim())}`)
  }

  return (
    <section className="brand-gradient pt-14 pb-10 px-5">
      <AnimatedHeadline />
      
      <p className="text-center text-white/80 text-sm mt-3 mb-6">
        Nigerian · Ghanaian · Jamaican · Caribbean
      </p>

      {/* Postcode input */}
      <div className="bg-white rounded-2xl p-1.5 flex items-center gap-2 shadow-card-lg max-w-sm mx-auto">
        <div className="flex items-center gap-2 flex-1 px-3">
          <MapPin className="w-4 h-4 text-brand flex-shrink-0" />
          <input
            type="text"
            placeholder="Enter your postcode"
            value={postcode}
            onChange={e => { setPostcode(e.target.value.toUpperCase()); setError('') }}
            onKeyDown={e => e.key === 'Enter' && handleSubmit()}
            className="flex-1 text-dark placeholder-mid text-[15px] font-medium outline-none bg-transparent"
            maxLength={8}
            autoComplete="postal-code"
          />
        </div>
        <button
          onClick={handleSubmit}
          className="flex items-center gap-1.5 bg-brand text-white px-4 py-2.5 rounded-xl font-semibold text-sm hover:bg-brand-dark transition-colors touch-target"
        >
          Find food
          <ArrowRight className="w-4 h-4" />
        </button>
      </div>

      {error && (
        <p className="text-center text-white/90 text-xs mt-2">{error}</p>
      )}

      {/* Trust micro-signals */}
      <div className="flex items-center justify-center gap-6 mt-5">
        {[
          { icon: '🍽️', text: '50+ vendors' },
          { icon: '⭐', text: '4.8 avg rating' },
          { icon: '🔒', text: 'FSA verified' },
        ].map(({ icon, text }) => (
          <div key={text} className="flex items-center gap-1.5 text-white/80 text-xs">
            <span>{icon}</span>
            <span className="font-medium">{text}</span>
          </div>
        ))}
      </div>
    </section>
  )
}

Create apps/web/src/components/vendor/vendor-card.tsx.
This is the MOST IMPORTANT component in the entire PWA — every vendor listing uses it.

'use client'
import Link from 'next/link'
import Image from 'next/image'
import { Star, Clock, ShoppingBag, Shield } from 'lucide-react'
import { cn } from '@/lib/utils'
import type { Vendor } from '@feastpot/types'

interface VendorCardProps {
  vendor: Vendor
  compact?: boolean  // compact = carousel card (narrower)
}

export function VendorCard({ vendor, compact = false }: VendorCardProps) {
  const minOrder = vendor.minOrderPence ? `£${(vendor.minOrderPence / 100).toFixed(0)} min` : null

  return (
    <Link href={`/vendors/${vendor.slug}`} className="block group">
      <div className={cn(
        'fp-card overflow-hidden transition-transform duration-200 active:scale-[0.98]',
        compact ? 'w-48' : 'w-full'
      )}>
        {/* Cover image — the hero of the card */}
        <div className={cn('relative overflow-hidden', compact ? 'h-28' : 'h-40')}>
          {vendor.coverUrl ? (
            <Image
              src={vendor.coverUrl}
              alt={vendor.businessName}
              fill
              className="object-cover group-hover:scale-105 transition-transform duration-500"
              sizes={compact ? '192px' : '(max-width: 640px) 100vw, 480px'}
            />
          ) : (
            <div className="absolute inset-0 brand-gradient flex items-center justify-center">
              <span className="text-4xl">🍽️</span>
            </div>
          )}

          {/* Gradient overlay for text legibility */}
          <div className="absolute inset-0 bg-gradient-to-t from-dark/60 via-transparent to-transparent" />

          {/* Badges top-right */}
          <div className="absolute top-2 right-2 flex flex-col gap-1 items-end">
            {vendor.communityFavourite && (
              <span className="flex items-center gap-1 bg-brand text-white text-[10px] font-bold px-2 py-0.5 rounded-full shadow-sm">
                🔥 Community Fave
              </span>
            )}
            {vendor.fsaRating && vendor.fsaRating >= 4 && (
              <span className="flex items-center gap-1 bg-teal text-white text-[10px] font-bold px-2 py-0.5 rounded-full shadow-sm">
                <Shield className="w-2.5 h-2.5" />
                Hygiene {vendor.fsaRating}/5
              </span>
            )}
          </div>

          {/* Vendor logo overlay */}
          {vendor.logoUrl && (
            <div className="absolute bottom-2 left-2 w-9 h-9 rounded-xl overflow-hidden border-2 border-white shadow-sm bg-white">
              <Image src={vendor.logoUrl} alt="" fill className="object-cover" />
            </div>
          )}
        </div>

        {/* Card body */}
        <div className={cn('bg-white', compact ? 'p-2.5' : 'p-3')}>
          <div className="flex items-start justify-between gap-2">
            <h3 className={cn('font-bold text-dark leading-tight', compact ? 'text-sm' : 'text-[15px]')}>
              {vendor.businessName}
            </h3>
            <div className="flex items-center gap-0.5 flex-shrink-0">
              <Star className="w-3 h-3 fill-amber-400 text-amber-400" />
              <span className="text-xs font-bold text-dark">{vendor.avgRating?.toFixed(1)}</span>
              {!compact && (
                <span className="text-[10px] text-mid">({vendor.reviewCount})</span>
              )}
            </div>
          </div>

          {/* Cuisine tags */}
          <div className="flex flex-wrap gap-1 mt-1.5">
            {vendor.cuisineTypes?.slice(0, compact ? 1 : 2).map(c => (
              <span key={c} className="text-[10px] text-mid bg-surface px-1.5 py-0.5 rounded-md">
                {c}
              </span>
            ))}
          </div>

          {/* Meta row */}
          {!compact && (
            <div className="flex items-center gap-3 mt-2">
              {minOrder && (
                <div className="flex items-center gap-1 text-[10px] text-mid">
                  <ShoppingBag className="w-3 h-3" />
                  {minOrder}
                </div>
              )}
              <div className="flex items-center gap-1 text-[10px] text-mid">
                <Clock className="w-3 h-3" />
                Scheduled delivery
              </div>
            </div>
          )}
        </div>
      </div>
    </Link>
  )
}
