getBalance(userId: string): Promise<number>
  SELECT SUM(points) FROM loyalty_points WHERE user_id = userId
  Use: await this.prisma.loyaltyPoint.aggregate({ where: { userId }, _sum: { points: true } })
  Return: _sum.points ?? 0

async creditPoints(userId: string, orderId: string, orderTotalPence: number): Promise<void>
  const points = Math.floor(orderTotalPence / 100)  // 1 point per £1
  const currentBalance = await this.getBalance(userId)
  const balanceAfter = currentBalance + points
  const expiresAt = new Date(Date.now() + 365 * 24 * 60 * 60 * 1000) // 12 months
  await this.prisma.loyaltyPoint.create({
    data: { userId, orderId, transactionType: 'earned', points, balanceAfter,
            description: 'Earned from order', expiresAt }
  })

async redeemPoints(userId: string, pointsToRedeem: number, orderId: string): Promise<number>
  Validate: if pointsToRedeem < 200 throw BadRequestException('Minimum redemption is 200 points')
  const balance = await this.getBalance(userId)
  if (balance < pointsToRedeem) throw BadRequestException('Insufficient points')
  const discountPence = pointsToRedeem  // 1 point = 1p
  const balanceAfter = balance - pointsToRedeem
  await this.prisma.loyaltyPoint.create({
    data: { userId, orderId, transactionType: 'redeemed', points: -pointsToRedeem,
            balanceAfter, description: `Redeemed ${pointsToRedeem} points at checkout` }
  })
  return discountPence

async expirePoints(): Promise<void>
  // Called by nightly cron @Cron('30 0 * * *')
  const now = new Date()
  const expiring = await this.prisma.loyaltyPoint.findMany({
    where: { transactionType: 'earned', expiresAt: { lt: now }, points: { gt: 0 } }
  })
  for (const row of expiring) {
    const balance = await this.getBalance(row.userId)
    await this.prisma.loyaltyPoint.create({
      data: { userId: row.userId, transactionType: 'expired',
              points: -row.points, balanceAfter: Math.max(0, balance - row.points),
              description: 'Points expired after 12 months' }
    })
    // Notify user
    await this.notificationsQueue.add('send_email', {
      to: row.userId, template: 'points_expiry',
      data: { points: row.points }
    })
  }

async adjustPoints(userId: string, points: number, reason: string, adminUserId: string): Promise<void>
  const balance = await this.getBalance(userId)
  await this.prisma.loyaltyPoint.create({
    data: { userId, transactionType: 'adjusted', points,
            balanceAfter: balance + points, description: reason }
  })
  await this.prisma.auditLog.create({
    data: { actorUserId: adminUserId, entityType: 'users', entityId: userId,
            action: 'loyalty.adjusted', newState: { points, reason } }
  })

async getHistory(userId: string, take = 20, cursor?: string)
  return this.prisma.loyaltyPoint.findMany({
    where: { userId }, orderBy: { createdAt: 'desc' },
    take, cursor: cursor ? { id: cursor } : undefined
  })

generateCode(): string
  return Math.random().toString(36).substring(2, 10).toUpperCase()

async ensureReferralCode(userId: string): Promise<string>
  const user = await this.prisma.user.findUnique({ where: { id: userId } })
  if (user.referralCode) return user.referralCode
  const code = this.generateCode()
  await this.prisma.user.update({ where: { id: userId }, data: { referralCode: code } })
  return code

async processReferral(newUserId: string, referralCode: string): Promise<void>
  const referrer = await this.prisma.user.findUnique({ where: { referralCode } })
  if (!referrer || referrer.id === newUserId) return  // ignore invalid/self-referral
  await this.prisma.referral.create({
    data: { referrerUserId: referrer.id, referredUserId: newUserId,
            referralCode, status: 'pending' }
  })

async rewardReferral(newUserId: string): Promise<void>
  // Called when new user's first order is delivered
  const referral = await this.prisma.referral.findUnique({
    where: { referredUserId: newUserId }, include: { referrer: true }
  })
  if (!referral || referral.status !== 'pending') return
  // Credit both users 500 points (= £5)
  await this.loyaltyService.adjustPoints(referral.referrerUserId, 500,
    'Referral reward — friend placed first order', 'system')
  await this.loyaltyService.adjustPoints(newUserId, 500,
    'Welcome bonus — joined via referral', 'system')
  await this.prisma.referral.update({
    where: { id: referral.id },
    data: { status: 'rewarded', rewardCreditPence: 500 }
  })
  // Notify both
  await this.notificationsQueue.add('send_email', {
    to: referral.referrer.email, template: 'referral_rewarded',
    data: { referredName: 'your friend', creditAmount: '£5' }
  })

@Get('loyalty-points')
async getMyLoyalty(@CurrentUser() user) {
  const [balance, history] = await Promise.all([
    this.loyaltyService.getBalance(user.id),
    this.loyaltyService.getHistory(user.id)
  ])
  return { balance, worthPence: balance, history }
}

@Get('referrals')
async getMyReferrals(@CurrentUser() user) {
  const code = await this.referralService.ensureReferralCode(user.id)
  const referrals = await this.prisma.referral.findMany({
    where: { referrerUserId: user.id }, orderBy: { createdAt: 'desc' }
  })
  return {
    referralCode: code,
    shareUrl: `https://feastpot.co.uk/join?ref=${code}`,
    referrals,
    totalEarnedPence: referrals.filter(r => r.status === 'rewarded').length * 500
  }
}
