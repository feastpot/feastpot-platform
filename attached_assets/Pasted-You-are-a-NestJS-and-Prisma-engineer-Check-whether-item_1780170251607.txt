IMPLEMENTATION (only if photo ordering is not wired)
════════════════════════════════════════════════════════

IMPORTANT: The audit confirmed imageUrls is a flat String[] with no
sort field. This means the cleanest fix without a schema migration
is to treat array position as sort order — the first image is always
the cover. Drag reorder simply rewrites the array in the new order.

STEP 1 — API: PATCH image order endpoint
File: apps/api/src/modules/catalogue/catalogue.controller.ts

@Patch(':vendorId/menus/:menuId/items/:itemId/images/reorder')
@Roles('vendor', 'admin')
async reorderItemImages(
  @Param('itemId', ParseUUIDPipe) itemId: string,
  @Body() dto: { imageUrls: string[] },
  @CurrentUser() user: JwtUser,
) {
  return this.catalogueService.reorderItemImages(itemId, dto.imageUrls)
}

File: apps/api/src/modules/catalogue/catalogue.service.ts

async reorderItemImages(itemId: string, imageUrls: string[]) {
  return this.prisma.menuItem.update({
    where: { id: itemId },
    data:  { imageUrls },
  })
}
