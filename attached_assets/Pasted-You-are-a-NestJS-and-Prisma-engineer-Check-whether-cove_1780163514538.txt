async registerInterest(dto: { email: string; postcode: string; source?: string }) {
  // Check for duplicate before inserting
  const existing = await this.prisma.coverageInterest.findFirst({
    where: { email: dto.email, postcode: dto.postcode.toUpperCase().trim() },
  })
  if (existing) {
    this.logger.log(`[coverage-interest] duplicate: ${dto.email} / ${dto.postcode}`)
    return { registered: true, duplicate: true }
  }

  await this.prisma.coverageInterest.create({
    data: {
      email:   dto.email,
      postcode: dto.postcode.toUpperCase().trim(),
      source:  dto.source ?? 'coverage-check',
    },
  })

  this.logger.log(`[coverage-interest] saved: ${dto.email} / ${dto.postcode}`)
  return { registered: true, duplicate: false }
}

@Get('admin/coverage-interest/count')
@Roles('admin', 'support')
async getCoverageInterestCount() {
  const count = await this.prisma.coverageInterest.count()
  const byPostcode = await this.prisma.coverageInterest.groupBy({
    by: ['postcode'],
    _count: { postcode: true },
    orderBy: { _count: { postcode: 'desc' } },
    take: 10,
  })
  return { total: count, topPostcodes: byPostcode }
}
