Create a payout reconciliation check script for ops use:

File: scripts/verify-payout-reconciliation.ts

This script connects to the database and Stripe and confirms:
1. Every payout with status 'transferred' has a real Stripe transfer
2. Every Stripe transfer amount matches the DB payout amount
3. No vendor is owed money that has not been paid

import { PrismaClient } from '@prisma/client'
import Stripe from 'stripe'

const prisma = new PrismaClient()
const stripe = new Stripe(process.env.STRIPE_SECRET_KEY!, { apiVersion: '2024-06-20' })

async function verifyPayoutReconciliation() {
  console.log('\n════════ PAYOUT RECONCILIATION CHECK ════════')

  const payouts = await prisma.payout.findMany({
    where:   { status: 'transferred' },
    include: { vendor: { select: { businessName: true } } },
    orderBy: { transferredAt: 'desc' },
    take:    50,
  })

  console.log(`\nChecking ${payouts.length} transferred payouts...\n`)

  let discrepancies = 0
  for (const payout of payouts) {
    if (!payout.stripeTransferId) {
      console.error(`❌ ${payout.vendor.businessName}: status=transferred but no stripeTransferId`)
      discrepancies++
      continue
    }
    try {
      const transfer = await stripe.transfers.retrieve(payout.stripeTransferId)
      if (transfer.amount !== payout.amountPence) {
        console.error(
          `❌ ${payout.vendor.businessName}: DB amount=${payout.amountPence}p ` +
          `Stripe amount=${transfer.amount}p — DISCREPANCY`
        )
        discrepancies++
      } else {
        console.log(`✅ ${payout.vendor.businessName}: £${(payout.amountPence/100).toFixed(2)} confirmed`)
      }
    } catch (e) {
      console.error(`❌ ${payout.vendor.businessName}: Stripe lookup failed — ${e.message}`)
      discrepancies++
    }
  }

  console.log(`\n════════ RESULT: ${discrepancies === 0 ? '✅ ZERO DISCREPANCIES' : `❌ ${discrepancies} DISCREPANCIES FOUND`} ════════\n`)
  await prisma.$disconnect()
  process.exit(discrepancies > 0 ? 1 : 0)
}

verifyPayoutReconciliation().catch(e => {
  console.error(e)
  process.exit(1)
})
