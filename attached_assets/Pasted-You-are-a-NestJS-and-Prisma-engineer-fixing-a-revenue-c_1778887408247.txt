STEP 5 — Verify payout calculation

it('calculates 12% commission correctly', () => {
  const grossPence = 10000  // £100
  const commissionBps = 1200
  const commissionPence = Math.round(grossPence * commissionBps / 10_000)
  const netPence = grossPence - commissionPence
  expect(commissionPence).toBe(1200)  // £12

it('calculates correctly on non-round amounts', () => {

})
