private getServiceFeePence(subtotalPence: number): number {
  const bps = parseInt(process.env.SERVICE_FEE_BPS ?? '0', 10)
  if (isNaN(bps) || bps < 0) return 0
  return Math.round(subtotalPence * bps / 10_000)
}

'use client'
import { useEffect, useState } from 'react'
import {
  useStripe,
  PaymentRequestButtonElement,
} from '@stripe/react-stripe-js'

export function AppleGooglePayButton({
  totalPence,
  onPaymentMethod,
}: {
  totalPence: number
  onPaymentMethod: (paymentMethodId: string) => void
}) {
  const stripe = useStripe()
  const [paymentRequest, setPaymentRequest] = useState<any>(null)
  const [available, setAvailable] = useState(false)

  useEffect(() => {
    if (!stripe || totalPence <= 0) return

    const pr = stripe.paymentRequest({
      country:  'GB',
      currency: 'gbp',
      total: {
        label:  'Feastpot order',
        amount: totalPence,
      },
      requestPayerName:  true,
      requestPayerEmail: true,
    })

    pr.canMakePayment().then(result => {
      if (result) {
        setPaymentRequest(pr)
        setAvailable(true)
      }
    })

    pr.on('paymentmethod', async (e) => {
      onPaymentMethod(e.paymentMethod.id)
      e.complete('success')
    })
  }, [stripe, totalPence])

  if (!available) return null

  return (
    <div style={{ marginBottom: '12px' }}>
      <PaymentRequestButtonElement
        options={{
          paymentRequest,
          style: {
            paymentRequestButton: {
              type:   'default',
              theme:  'dark',
              height: '48px',
            },
          },
        }}
      />
      <div style={{
        display: 'flex', alignItems: 'center', gap: '10px', margin: '12px 0',
      }}>
        <div style={{ flex: 1, height: '1px', background: '#E8E8E8' }} />
        <span style={{ fontSize: '12px', color: '#999999' }}>or pay by card</span>
        <div style={{ flex: 1, height: '1px', background: '#E8E8E8' }} />
      </div>
    </div>
  )
}
