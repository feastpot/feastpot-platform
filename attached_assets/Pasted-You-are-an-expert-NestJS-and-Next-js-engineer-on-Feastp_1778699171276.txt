async proposeAmendment(orderId: string, vendorUserId: string, dto: ProposeAmendmentDto) {
  const vendor = await this.prisma.vendor.findUnique({ where: { userId: vendorUserId } })
  const order = await this.prisma.order.findUnique({
    where: { id: orderId }, include: { customer: { select: { email: true } } }
  })
  if (!order) throw new NotFoundException('Order not found')
  if (order.vendorId !== vendor.id) throw new ForbiddenException('Not your order')
  if (![OrderStatus.accepted, OrderStatus.preparing].includes(order.status))
    throw new BadRequestException('Order cannot be amended at this stage')

  // Check no pending amendment already exists
  const existing = await this.prisma.orderAmendment.findFirst({
    where: { orderId, status: 'pending' }
  })
  if (existing) throw new ConflictException('An amendment is already pending for this order')

  const expiresAt = new Date(Date.now() + 15 * 60 * 1000)
  const amendment = await this.prisma.orderAmendment.create({
    data: { orderId, vendorId: vendor.id, proposedChange: dto.proposedChange,
            priceDeltaPence: dto.priceDeltaPence ?? 0, expiresAt }
  })

  // Enqueue expiry job
  await this.ordersQueue.add('expire_amendment', { amendmentId: amendment.id },
    { delay: 15 * 60 * 1000, jobId: `expire-amendment-${amendment.id}` })

  // Notify customer
  await this.notificationsQueue.add('send_email', {
    to: order.customer.email, template: 'order_amendment_proposed',
    data: { proposedChange: dto.proposedChange, orderId, priceDelta: dto.priceDeltaPence ?? 0 }
  })
  return amendment
}

Replace the NotImplementedException in respondToAmendment() with:

async respondToAmendment(orderId: string, customerUserId: string, dto: RespondAmendmentDto) {
  const order = await this.prisma.order.findUnique({ where: { id: orderId } })
  if (!order) throw new NotFoundException()
  if (order.customerId !== customerUserId) throw new ForbiddenException()

  const amendment = await this.prisma.orderAmendment.findFirst({
    where: { orderId, status: 'pending' }
  })
  if (!amendment) throw new NotFoundException('No pending amendment found')

  const newStatus: AmendmentStatus = dto.accepted ? 'accepted' : 'declined'
  await this.prisma.orderAmendment.update({
    where: { id: amendment.id },
    data: { status: newStatus, respondedAt: new Date() }
  })

  // Remove expiry job
  const job = await this.ordersQueue.getJob(`expire-amendment-${amendment.id}`)
  if (job) await job.remove()

  if (dto.accepted && amendment.priceDeltaPence < 0) {
    // Issue partial refund for price reduction
    await this.paymentsService.createRefund({
      orderId, amountPence: Math.abs(amendment.priceDeltaPence),
      reason: 'Order amendment — price reduction', disputeId: undefined
    }, 'system')
  }

  // Notify vendor
  await this.notificationsQueue.add('send_email', {
    to: 'vendor-notification', template: dto.accepted ? 'amendment_accepted' : 'amendment_declined',
    data: { orderId }
  })
  return { status: newStatus }
}
