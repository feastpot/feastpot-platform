PART 3 — Redis caching: cache the 3 highest-volume reads
═══════════════════════════════════════════════════════

File: apps/api/src/modules/vendors/vendors.service.ts

At 10,000 orders/day, vendor browse is called ~50,000 times/day.
Without caching, that is 50,000 DB queries just for browse.
With 5-minute cache: 50,000 → ~17 DB queries (one per cache miss per 5 min window).

Add caching to searchVendors, findById, and menu fetching:

async searchVendors(dto: SearchVendorsDto) {
  const cacheKey = `vendors:search:${JSON.stringify(dto)}`
  const cached = await this.cache.get<VendorSearchResult>(cacheKey)
  if (cached) return cached

  const result = await this.vendorsRepository.search(dto)
  await this.cache.set(cacheKey, result, 300)  // 5 min TTL
  return result
}

async getVendorProfile(id: string) {
  const cacheKey = `vendors:profile:${id}`
  const cached = await this.cache.get<Vendor>(cacheKey)
  if (cached) return cached

  const vendor = await this.vendorsRepository.findById(id)
  await this.cache.set(cacheKey, vendor, 600)  // 10 min TTL
  return vendor
}

// Invalidate on any vendor update
async updateVendor(id: string, dto: UpdateVendorDto) {
  const result = await this.vendorsRepository.update(id, dto)
  await this.cache.del(`vendors:profile:${id}`)
  await this.cache.delByPattern('vendors:search:*')  // all search caches
  return result
}

// Menu items: short TTL because availability can change instantly
async getVendorMenus(vendorId: string) {
  const cacheKey = `vendors:menus:${vendorId}`
  const cached = await this.cache.get(cacheKey)
  if (cached) return cached

  const menus = await this.catalogueRepository.findByVendor(vendorId)
  await this.cache.set(cacheKey, menus, 120)  // 2 min TTL
  return menus
}

// When vendor toggles item availability: immediate cache invalidation
async toggleItemAvailability(vendorId: string, itemId: string, isAvailable: boolean) {
  await this.prisma.menuItem.update({ where: { id: itemId }, data: { isAvailable } })
  await this.cache.del(`vendors:menus:${vendorId}`)  // immediate — sold-out must show NOW
}

PART 6 — Multi-instance throttler: swap to Redis storage
═══════════════════════════════════════════════════════

With 30 Autoscale instances, the default in-process Map throttler
counts requests per-instance, not globally. At 30 instances, effective
rate limit is 30× the configured value — discount codes could be brute-forced.

File: apps/api/src/app.module.ts

npm install @nest-lab/throttler-storage-redis -w @feastpot/api

ThrottlerModule.forRootAsync({
  imports: [ConfigModule],
  useFactory: (config: ConfigService) => ({
    storage: new ThrottlerStorageRedisService(
      new Redis(config.get('REDIS_URL'))
    ),
    throttlers: [
      { name: 'short', ttl: 1000,   limit: 20  },
      { name: 'long',  ttl: 60_000, limit: 300 },
    ],
  }),
  inject: [ConfigService],
})

This makes rate limits accurate across all 30 instances.
The Redis-backed throttler adds ~2ms per rate-checked request (acceptable).

LoggerModule.forRoot({
  pinoHttp: {
    level: process.env.NODE_ENV === 'production' ? 'info' : 'debug',
    transport: process.env.NODE_ENV !== 'production'
      ? { target: 'pino-pretty', options: { colorize: true } }
      : undefined,
    serializers: {
      req: (req) => ({ method: req.method, url: req.url, requestId: req.id }),
      res: (res) => ({ statusCode: res.statusCode }),
    },
    customProps: (req) => ({
      userId: (req as any).user?.id,
      role:   (req as any).user?.role,
    }),
    redact: ['req.headers.authorization', 'req.headers.cookie'],
  },
})
