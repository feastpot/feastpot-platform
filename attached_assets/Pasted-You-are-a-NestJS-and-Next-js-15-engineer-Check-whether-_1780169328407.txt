@Patch(':vendorId/menus/reorder')
@Roles('vendor', 'admin')
async reorderMenus(
  @Param('vendorId', ParseUUIDPipe) vendorId: string,
  @Body() dto: { menuIds: string[] },
  @CurrentUser() user: JwtUser,
) {
  await this.catalogueService.verifyVendorOwnership(vendorId, user.id)
  return this.catalogueService.reorderMenus(vendorId, dto.menuIds)
}

File: apps/api/src/modules/catalogue/catalogue.service.ts

async reorderMenus(vendorId: string, menuIds: string[]) {
  await this.prisma.$transaction(
    menuIds.map((id, index) =>
      this.prisma.menu.update({
        where: { id, vendorId },
        data:  { displayOrder: index },
      })
    )
  )
  return this.prisma.menu.findMany({
    where:   { vendorId },
    orderBy: { displayOrder: 'asc' },
  })
}
