@Cron('0 */6 * * *')  // every 6 hours
async expireStaleEnquiries() {
  const SLA_HOURS = 48  // vendor must quote within 48h of enquiry submission
  const expiryThreshold = new Date(Date.now() - SLA_HOURS * 60 * 60 * 1000)

  const stale = await this.prisma.eventEnquiry.findMany({
    where: {
      status: EnquiryStatus.open,
      createdAt: { lt: expiryThreshold },
    },
    include: { customer: { select: { email: true } } },
  })

  for (const enquiry of stale) {
    await this.prisma.eventEnquiry.update({
      where: { id: enquiry.id },
      data: { status: EnquiryStatus.expired },
    })
    // Notify customer
    await this.notificationsQueue.add('send_email', {
      to: enquiry.customer.email,
      template: 'enquiry_expired',
      data: { enquiryId: enquiry.id },
    })
  }
  this.logger.log(`[EnquiriesCron] Expired ${stale.length} stale enquiries`)
}
