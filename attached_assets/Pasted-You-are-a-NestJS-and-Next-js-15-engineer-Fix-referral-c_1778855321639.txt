PART 1 — API: validate endpoint
File: apps/api/src/modules/loyalty/loyalty.controller.ts

@Get('referrals/validate')
@Public()
async validateCode(@Query('code') code: string) {
  if (!code || code.length < 4) throw new BadRequestException('Invalid code')
  const user = await this.prisma.user.findFirst({
    where: { referralCode: { equals: code, mode: 'insensitive' } },
    select: { id: true, profile: { select: { firstName: true } } },
  })
  if (!user) throw new NotFoundException('Referral code not found or expired')
  return { valid: true, referrerFirstName: user.profile?.firstName || 'a Feastpot member', bonusPence: 500 }
}

useEffect(() => {
  if (!code) { router.replace('/register'); return }
  fetch(`${process.env.NEXT_PUBLIC_API_URL}/v1/referrals/validate?code=${code}`)
    .then(r => r.json())
    .then(data => {
      if (data.valid) {
        localStorage.setItem('fp.referral_code', code.toUpperCase())
        setReferrerName(data.referrerFirstName)
        setState('valid')
        setTimeout(() => router.replace('/register'), 2000)
      } else { setState('invalid') }
    })
    .catch(() => setState('invalid'))
}, [code])
