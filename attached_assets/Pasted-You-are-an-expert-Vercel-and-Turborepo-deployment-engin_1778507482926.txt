STEP 2 — Update next.config.ts in each app
Ensure each app's next.config.ts does NOT hardcode any API URL.
All API calls must use process.env.NEXT_PUBLIC_API_URL.

Create a shared API client at each app's src/lib/api.ts:
const API_BASE = process.env.NEXT_PUBLIC_API_URL ?? 'http://localhost:3001'
export const api = {
  get: (path: string, token?: string) =>
    fetch(`${API_BASE}${path}`, { headers: token ? { Authorization: `Bearer ${token}` } : {} }),
  post: (path: string, body: unknown, token?: string) =>
    fetch(`${API_BASE}${path}`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json', ...(token ? { Authorization: `Bearer ${token}` } : {}) },
      body: JSON.stringify(body),
    }),
}
