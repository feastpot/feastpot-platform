STEP 1 — Install and configure next-sitemap:
npm install next-sitemap
Create next-sitemap.config.js:
{
  siteUrl: 'https://feastpot.co.uk',
  generateRobotsTxt: true,
  exclude: ['/account/*', '/checkout', '/orders/*'],
  additionalPaths: async (config) => {
    // Fetch all live vendor slugs from API and add to sitemap
    const vendors = await fetch('https://api.feastpot.co.uk/v1/vendors?limit=1000&status=live').then(r => r.json())
    return vendors.data.map(v => ({ loc: '/vendors/' + v.slug, lastmod: v.updatedAt }))
  }
}
Add to apps/web/next.config.ts postbuild: "next-sitemap"
